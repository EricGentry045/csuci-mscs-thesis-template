\newglossaryentry{seaker}{
    name={SEAKER},
    description={A digital forensics triage device built using a raspberry pi.
    It is an acronym for Storage Evaluator And Knowledge Extraction Reader}
}

\newglossaryentry{rpi}{
    name={Raspberry Pi},
    description={A small, affordable computer created with the intention
    of providing low-cost computing power to the masses}
}

\newglossaryentry{ip}{
    name={IP},
    description={An Internet Protocol address that identifies a particular
    network connection for a computer}
}

\newglossaryentry{searchwarrant}{
    name={search warrant},
    description={A legal document authorizing a law enforcement official
    to enter and search a specific premises for specific things}
}

\newglossaryentry{wifi}{
    name={WIFI},
    description={A wireless network or access point allowing computers,
    smart phones, and other devices to connect with each other and
    the Internet}
}

\newglossaryentry{usb}{
    name={USB},
    description={An input/output standard for a cable interface and
    protocols between computers and devices (Universal Serial Bus)}
}

\newglossaryentry{sata}{
    name={SATA},
    description={An input/output standard for connecting the computer
    bus interface to mass storage devices, like hard drives
    (Serial Advanced Technology Attachment)}
}

\newglossaryentry{ssh}{
    name={SSH},
    description={A software program that enables encrypted, command-line
    communication with another networked computer.
    (Secure Shell)}
}
